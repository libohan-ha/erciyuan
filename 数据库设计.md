1.users集合

{
  _id: ObjectId, // MongoDB自动生成的唯一ID
  username: String, // 用户名，唯一
  password: String, // 存储加密后的密码
  avatarUrl: String, // 头像图片的URL，可选
  // ... 其他用户信息
}

2.images


 {
  _id: ObjectId,
  url: String,        // 图片的访问URL (例如: /uploads/xxx.jpg)
  originalName: String, // 原始文件名
  title: String,      // 图片标题 (比描述更突出)
  description: String, // 图片描述，可选
  tags: [String],     // 标签数组，如 ["初音未来", "Vocaloid", "蓝色头发"]
  ownerId: ObjectId,  // 图片所有者ID，关联到 users 集合
  createdAt: Date,    // 上传时间
  albumId: ObjectId   // 所属相册ID，可选，关联到 albums 集合
}

3.相册集合：

{
  _id: ObjectId,
  name: String,       // 相册名称，如 "我的老婆们"
  description: String,// 相册描述
  ownerId: ObjectId,  // 相册所有者ID
  coverImageId: ObjectId, // 相册封面图片ID，可选
  createdAt: Date,
}
